UpdateSequence ::= UpdateContainer ( <SEMICOLON> ( UpdateContainer )? )* <EOF>
UpdateContainer ::= Prolog Update
QueryContainer ::= Prolog Query <EOF>
Prolog ::= ( PrefixDecl | BaseDecl )*
BaseDecl ::= <BASE> <Q_IRI_REF>
PrefixDecl ::= <PREFIX> <PNAME_NS> IRI
Query ::= ( SelectQuery | ConstructQuery | DescribeQuery | AskQuery )
SelectQuery ::= Select ( DatasetClause )* WhereClause SolutionModifier ( BindingsClause )?
Select ::= <SELECT> ( <DISTINCT> | <REDUCED> )? ( <STAR> | ( ProjectionElem )+ )
ProjectionElem ::= ( Var | <LPAREN> Expression <AS> Var <RPAREN> )
ConstructQuery ::= Construct ( DatasetClause )* WhereClause SolutionModifier ( BindingsClause )?
Construct ::= <CONSTRUCT> ( ConstructClause )?
ConstructClause ::= <LBRACE> ( TriplesBlock )? <RBRACE>
DescribeQuery ::= Describe ( DatasetClause )* ( WhereClause )? SolutionModifier ( BindingsClause )?
Describe ::= <DESCRIBE> ( <STAR> | ( VarOrIRIref )+ )
AskQuery ::= <ASK> ( DatasetClause )* WhereClause ( BindingsClause )?
DatasetClause ::= <FROM> ( <NAMED> )? IRIref
WhereClause ::= ( <WHERE> )? GroupGraphPattern
BindingsClause ::= <BINDINGS> ( Var )* <LBRACE> ( BindingSet )* <RBRACE>
BindingSet ::= ( <LPAREN> ( BindingValue )* <RPAREN> | <NIL> )
BindingValue ::= ( IRIref | RDFLiteral | NumericLiteral | BooleanLiteral | <UNDEF> )
SolutionModifier ::= ( GroupClause )? ( HavingClause )? ( OrderClause )? ( LimitOffsetClauses )?
GroupClause ::= <GROUP> <BY> ( GroupCondition )+
OrderClause ::= <ORDER> <BY> ( OrderCondition )+
GroupCondition ::= ( FunctionCall | BuiltInCall | <LPAREN> Expression ( <AS> Var )? <RPAREN> | Var )
HavingClause ::= <HAVING> Constraint
OrderCondition ::= ( ( <ASC> | <DESC> )? BrackettedExpression | FunctionCall | BuiltInCall | Var )
LimitOffsetClauses ::= Limit ( Offset )?
| Offset ( Limit )?
Limit ::= <LIMIT> <INTEGER>
Offset ::= <OFFSET> <INTEGER>
GroupGraphPattern ::= <LBRACE> ( SelectQuery | GraphPattern ) <RBRACE>
GraphPattern ::= ( BasicGraphPattern )? ( GraphPatternNotTriples ( <DOT> )? GraphPattern )?
BasicGraphPattern ::= ( TriplesBlock ( FilterOrBind ( <DOT> )? ( TriplesBlock )? )* | ( FilterOrBind ( <DOT> )? ( TriplesBlock )? )+ )
FilterOrBind ::= Filter
| Bind
TriplesBlock ::= TriplesSameSubjectPath ( <DOT> TriplesSameSubjectPath )* ( <DOT> )?
TriplesTemplate ::= TriplesSameSubject ( <DOT> TriplesSameSubject )* ( <DOT> )?
GraphPatternNotTriples ::= OptionalGraphPattern
| GroupOrUnionGraphPattern
| GraphGraphPattern
| MinusGraphPattern
| ServiceGraphPattern
OptionalGraphPattern ::= <OPTIONAL> <LBRACE> ( SelectQuery | GraphPattern ) <RBRACE>
GraphGraphPattern ::= <GRAPH> VarOrIRIref GroupGraphPattern
GroupOrUnionGraphPattern ::= GroupGraphPattern ( <UNION> GroupOrUnionGraphPattern )?
MinusGraphPattern ::= <MINUS_SETOPER> GroupGraphPattern
ServiceGraphPattern ::= <SERVICE> ( <SILENT> )? VarOrIRIref GroupGraphPattern
Filter ::= <FILTER> Constraint
Constraint ::= ( BrackettedExpression | BuiltInCall | FunctionCall )
FunctionCall ::= IRIref ArgList
ArgList ::= <NIL>
| <LPAREN> Expression ( <COMMA> Expression )* <RPAREN>
TriplesSameSubject ::= ( VarOrTerm PropertyList | TriplesNode ( PropertyList )? )
PropertyList ::= Verb ObjectList ( <SEMICOLON> ( PropertyList )? )?
ObjectList ::= Object ( <COMMA> Object )*
TriplesSameSubjectPath ::= ( VarOrTerm PropertyListPath | TriplesNode ( PropertyListPath )? )
PropertyListPath ::= ( VerbPath | VerbSimple ) ObjectList ( <SEMICOLON> ( PropertyListPath )? )*
VerbPath ::= Path
Path ::= PathAlternative
PathAlternative ::= PathSequence ( <PIPE> PathSequence )*
PathSequence ::= PathElt ( <SLASH> PathElt )*
PathElt ::= ( <INVERSE> )? PathPrimary ( PathMod )?
PathPrimary ::= IRIref
| <IS_A>
| <NOT> PathNegatedPropertySet
| <LPAREN> Path <RPAREN>
PathNegatedPropertySet ::= PathOneInPropertySet
| <LPAREN> ( PathOneInPropertySet ( <PIPE> PathOneInPropertySet )* )? <RPAREN>
PathOneInPropertySet ::= ( IRIref | <IS_A> | <INVERSE> ( IRIref | <IS_A> ) )
PathMod ::= ( <STAR> | <QUESTION> | <PLUS> | <LBRACE> ( <INTEGER> ( <COMMA> ( <RBRACE> | <INTEGER> <RBRACE> ) | <RBRACE> ) | <COMMA> <INTEGER> <RBRACE> ) )
VerbSimple ::= Var
Object ::= GraphNode
Verb ::= VarOrIRIref
| <IS_A>
TriplesNode ::= Collection
| BlankNodePropertyList
BlankNodePropertyList ::= <LBRACK> PropertyList <RBRACK>
Collection ::= <LPAREN> ( GraphNode )+ <RPAREN>
GraphNode ::= VarOrTerm
| TriplesNode
VarOrTerm ::= Var
| GraphTerm
VarOrIRIref ::= Var

Var ::= ( <VAR1> | <VAR2> )

ConditionalOrExpression ::= ConditionalAndExpression ( <OR> ConditionalOrExpression )?

NumericExpression ::= AdditiveExpression
AdditiveExpression ::= MultiplicativeExpression ( <PLUS> MultiplicativeExpression | <MINUS> MultiplicativeExpression | NumericLiteralPositive | NumericLiteralNegative )*
MultiplicativeExpression ::= UnaryExpression ( <STAR> UnaryExpression | <SLASH> UnaryExpression )*
UnaryExpression ::= <NOT> PrimaryExpression

GroupConcat ::= <GROUP_CONCAT> <LPAREN> ( <DISTINCT> )? Expression ( <SEMICOLON> <SEPARATOR> <EQ> Expression )? <RPAREN>
